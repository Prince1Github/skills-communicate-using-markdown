```python
def calculate_factorial(n):
    """Calculate the factorial of a number."""
    if n < 0:
        return "Factorial is not defined for negative numbers"
    elif n == 0 or n == 1:
        return 1
    else:
        result = 1
        for i in range(2, n + 1):
            result *= i
        return result

# Example usage
print(f"5! = {calculate_factorial(5)}")
print(f"0! = {calculate_factorial(0)}")
```
